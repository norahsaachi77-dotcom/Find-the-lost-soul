  <p id="status">Redo.</p>

  <script>

    const statusEl = document.getElementById('status');

    const player = document.getElementById('player');

    async function openCamera(constraints) {

      statusEl.textContent = 'Begär kameratillstånd…';

      try {

        const stream = await navigator.mediaDevices.getUserMedia(constraints);

        player.srcObject = stream;

        statusEl.textContent = 'Kameran är igång.';

      } catch (err) {

        statusEl.textContent = 'Kunde inte öppna kameran: ' + err.name;

      }

    }

    document.getElementById('btnFront').onclick = () => openCamera({ video: { facingMode: { ideal: "user" } } });

    document.getElementById('btnBack').onclick  = () => openCamera({ video: { facingMode: { ideal: "environment" } } });

  </script>
